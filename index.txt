        Connect Device

    <div class="grid grid-cols-1 lg:grid-cols-3 gap-6">
      <div class="glass p-10 flex flex-col items-center justify-center text-center">
        <span class="text-slate-400 text-sm uppercase tracking-widest mb-4">Current Y-Angle</span>
        <div id="val" class="text-8xl font-black mb-2">0°</div>
        <div id="status-text" class="status-ok text-lg">Posture: Good</div>
      </div>

      <div class="lg:col-span-2 glass p-6">
        <canvas id="liveChart"></canvas>
      </div>
    </div>

    <div class="glass p-8">
      <h2 class="text-xl font-bold mb-6 flex items-center">
        <span class="w-2 h-6 bg-sky-500 rounded-full mr-3"></span> Daily Progress
      </h2>
      <div id="history-list" class="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-4 gap-4">
        </div>
    </div>
  </div>

  <script>
    let port, reader;
    const valDisplay = document.getElementById('val');
    const statusText = document.getElementById('status-text');
    const historyList = document.getElementById('history-list');

    // --- Chart Config ---
    const ctx = document.getElementById('liveChart').getContext('2d');
    const chart = new Chart(ctx, {
      type: 'line',
      data: {
        labels: [],
        datasets: [{
          label: 'Back Lean Angle (Y)',
          data: [],
          borderColor: '#38bdf8',
          backgroundColor: 'rgba(56, 189, 248, 0.1)',
          borderWidth: 3,
          fill: true,
          tension: 0.4,
          pointRadius: 0
        }]
      },
      options: {
        responsive: true,
        maintainAspectRatio: false,
        scales: {
          y: { min: 0, max: 90, grid: { color: '#1e293b' }, ticks: { color: '#64748b' } },
          x: { display: false }
        },
        plugins: { legend: { display: false } }
      }
    });

    // --- Bluetooth/Serial logic ---
    document.getElementById('btn-connect').onclick = async () => {
      try {
        port = await navigator.serial.requestPort();
        await port.open({ baudRate: 9600 });
        document.getElementById('btn-connect').innerText = "✅ Connected";
        readData();
      } catch (err) { console.log("Connection Failed", err); }
    };

    async function readData() {
      const decoder = new TextDecoderStream();
      port.readable.pipeTo(decoder.writable);
      reader = decoder.readable.getReader();
      let buffer = "";

      while (true) {
        const { value, done } = await reader.read();
        if (done) break;
        buffer += value;
        let lines = buffer.split('\n');
        buffer = lines.pop();

        for (let line of lines) {
          let num = parseFloat(line.trim());
          if (!isNaN(num)) {
            updateUI(num);
            saveToLocal(num);
          }
        }
      }
    }

    function updateUI(angle) {
      valDisplay.innerText = Math.round(angle) + "°";
      
      // อัปเดตกราฟ
      if (chart.data.labels.length > 30) {
        chart.data.labels.shift();
        chart.data.datasets[0].data.shift();
      }
      chart.data.labels.push("");
      chart.data.datasets[0].data.push(angle);
      chart.update('none');

      // เช็กท่าทาง
      if (angle > 60) {
        statusText.innerText = "Posture: Bad (Slumping!)";
        statusText.className = "status-warn text-lg";
      } else {
        statusText.innerText = "Posture: Good";
        statusText.className = "status-ok text-lg";
      }
    }

    // บันทึกประวัติ
    function saveToLocal(angle) {
      const today = new Date().toLocaleDateString();
      let history = JSON.parse(localStorage.getItem('postureData') || '{}');
      if (!history[today]) history[today] = [];
      
      // บันทึกเฉพาะค่าที่ก้มเกิน 60 (นับเป็นจำนวนครั้ง)
      if (angle > 60) {
        history[today].push(1); 
        localStorage.setItem('postureData', JSON.stringify(history));
        renderHistory();
      }
    }

    function renderHistory() {
      const history = JSON.parse(localStorage.getItem('postureData') || '{}');
      historyList.innerHTML = '';
      Object.keys(history).reverse().slice(0, 4).forEach(date => {
        const count = history[date].length;
        historyList.innerHTML += `
          <div class="bg-slate-800/50 p-4 rounded-2xl border border-slate-700">
            <div class="text-slate-400 text-xs">${date}</div>
            <div class="text-2xl font-bold text-sky-400">${count} <span class="text-sm font-normal text-slate-500">Alerts</span></div>
          </div>
        `;
      });
    }
    renderHistory();
  </script>
